CREATE TABLE users (
    id_user SERIAL PRIMARY KEY,
    nom VARCHAR(30),
    prenom VARCHAR(30),
    email VARCHAR(30),
    mdp VARCHAR(30),
    role VARCHAR(30)
);

CREATE TABLE reponses (
    id_user INT REFERENCES users(id_user),
    comptabilite INT,
    degre INT
);

CREATE TABLE tests (
    id_test SERIAL PRIMARY KEY,
    nom_test VARCHAR(30),
    description VARCHAR(50)
);
